# The most elegant search algorithm
def binary_search(arr, target):
    lo, hi = 0, len(arr) - 1

    while lo <= hi:
        mid = (lo + hi) // 2

        if arr[mid] == target:
            return mid # Found it!
        elif arr[mid] < target:
            lo = mid + 1
        else:
            hi = mid - 1

    return -1 # O(log n) time
